\begin{equation}

\Lang{Untuk lapisan tersembunyi ke-}{For the }$l$\Lang{, aktivasi neuron dihitung dengan}{-th hidden layer, neuron activations are computed by}
\begin{equation}

\end{equation}
\Lang{dengan}{where} $\mathbf{a}^{[0]}=\mathbf{x}$, $\mathbf{W}^{[l]}$
